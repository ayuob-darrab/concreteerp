المهمة 1: إنشاء PublicController
════════════════════════════════════════

أنشئ الملف: app/Http/Controllers/PublicController.php

    <?php

    namespace App\Http\Controllers;

    use App\Models\Company;
    use Illuminate\Http\Request;

    class PublicController extends Controller
    {
        // يجلب الشركة المالكة (SA) — يُستخدم في جميع الصفحات العامة
        private function getOwnerCompany()
        {
            return Company::where('code', 'SA')->first();
        }

        public function features()
        {
            $ownerCompany = $this->getOwnerCompany();
            return view('public.features', compact('ownerCompany'));
        }

        public function about()
        {
            $ownerCompany = $this->getOwnerCompany();
            return view('public.about', compact('ownerCompany'));
        }

        public function contact()
        {
            $ownerCompany = $this->getOwnerCompany();
            return view('public.contact', compact('ownerCompany'));
        }

        public function sitemap()
        {
            $pages = [
                ['url' => url('/'),                'priority' => '1.0', 'freq' => 'weekly'],
                ['url' => url('/system-benefits'), 'priority' => '0.9', 'freq' => 'weekly'],
                ['url' => url('/features'),        'priority' => '0.8', 'freq' => 'monthly'],
                ['url' => url('/about'),           'priority' => '0.7', 'freq' => 'monthly'],
                ['url' => url('/contact'),         'priority' => '0.6', 'freq' => 'monthly'],
            ];

            $content = view('public.sitemap', compact('pages'))->render();

            return response($content, 200)
                ->header('Content-Type', 'application/xml');
        }
    }

ملاحظة: جميع الدوال تمرر $ownerCompany (نفس نمط system-benefits)
منطق الواتساب يبقى في Blade وليس في الكونترولر — للتوحيد مع system-benefits.

المهمة 8: إضافة Navigation للصفحات العامة
════════════════════════════════════════

في ملف resources/views/layouts/auth.blade.php
أضف شريط تنقل بسيط داخل <body> قبل <div class="auth-container">

لذلك يجب إخفاء التنقل في صفحة تسجيل الدخول.

الطريقة المقترحة: استخدم @hasSection لعرض النافبار فقط إذا الصفحة عرّفت section('show_nav'):

    @hasSection('show_nav')

        <a href="{{ route('landing') }}">الرئيسية</a>
        <a href="{{ route('features') }}">المميزات</a>
        <a href="{{ route('about') }}">عن النظام</a>

    @endif

وفي كل صفحة عامة (system-benefits, features, about, contact) أضف:
    @section('show_nav', true)

هكذا صفحة /login لن تعرض النافبار لأنها لا تعرّف هذا الـ section.

[5] قواعد يجب اتباعها في كل ملف Blade عام
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. كل صفحة عامة يجب أن تمتد من layouts.auth وليس layouts.app
   @extends('layouts.auth')

2. كل صفحة يجب أن تحدد عنوانها الخاص:
   @section('title', 'عنوان الصفحة المحدد - ConcreteERP')

3. كل صفحة عامة تضيف: @section('show_nav', true)
   لعرض شريط التنقل (صفحة login لا تضيفه)

4. استخدم @push('page_meta') لإضافة وسوم SEO خاصة بالصفحة

5. كل صفحة يجب أن تحتوي على h1 واحد فقط يتضمن الكلمة المفتاحية الرئيسية

6. النصوص يجب أن تكون طبيعية وليست حشواً — اكتب كأنك تشرح النظام لعميل حقيقي

7. لا تضف dir="rtl" أو lang="ar" في الصفحات — القالب auth.blade.php يتكفل بذلك

8. لا تضع أي middleware auth على المسارات العامة

9. استخدم $ownerCompany (وليس $company) لبيانات الشركة المالكة

10. منطق تحويل رقم الواتساب يبقى في Blade (نفس نمط system-benefits)

11. استخدم route() وليس url() لتوليد الروابط — لضمان صحتها محلياً وفي الإنتاج

12. استخدم نفس تنسيق CSS من system-benefits (auth-card, auth-header, auth-body, إلخ)
    لتوحيد شكل الصفحات العامة
